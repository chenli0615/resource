        function static() {
            alert(foo);
        }
        (function() {
            var foo = 2;
            static();
        })();

        // 在js中，会弹出1而非2，因为static的scope在创建时，记录的foo是1。
        // 如果js是动态作用域，那么他应该弹出2。请体会一下两者的区别。
    </script>
